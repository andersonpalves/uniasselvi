    se(aprovacao(calculaMedia(n1,n2,n3)) == verdadeiro){
        escreva("Aprovado com: "+ calculaMedia(n1,n2,n3))
    }senao{
        escreva("Reprovado com: "+ calculaMedia(n1,n2,n3))
    }   
  }

 funcao real calculaMedia(real n1, real n2, real n3){
  real media = (n1 + n2 + n3) / 3
  retorne Matematica.arredondar(media,2)
 }
